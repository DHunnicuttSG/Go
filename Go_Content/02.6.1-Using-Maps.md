```go
	// List all
	fmt.Println("\nAll contacts:")
	for _, info := range contacts {
		fmt.Println(info.Name, "-", info.Phone, "-", info.Email)
	}

	// Delete one
	delete(contacts, "Maria")

	fmt.Println("\nAfter delete:")
	for _, info := range contacts {
		fmt.Println(info.Name)
	}
}
```
